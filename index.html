Code below calculates total length.

This number updates after change: 40. 

function getTotalLength(data){
    var result = 0;
    for (var i = 0; i < data.val.length; i++) {
      result += (data.val[i][1] - data.val[i][0]);
    }
    return result;
}

var result2el = document.getElementById('result2');
var result3el = document.getElementById('result3');

slider.on('changing', function (data) {
    result2el.innerHTML = getTotalLength(data).toString();
});

slider.on('change', function (data) {
    result3el.innerHTML = getTotalLength(data).toString();
});
